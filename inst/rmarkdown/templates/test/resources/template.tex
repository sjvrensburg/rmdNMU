\def\tick{
  \begin{tikzpicture}[scale=0.3]
    \draw[fill=black] (0,.35) -- (.25,0) -- (1,.7) -- (.25, .175) -- cycle;
    \node[above, white] at (0.35, 0.1) {\tiny 1};
    \node[below, white] at (0.7, 0.5) {\tiny 2};
  \end{tikzpicture}
}
